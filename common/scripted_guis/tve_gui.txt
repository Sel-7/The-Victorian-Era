check_if_has_universities_everywhere = {
	scope = country 
	is_valid = {
		NOT = {
			every_scope_state = {
				if = {
					limit = {
						is_incorporated = yes
					}
					OR = {
						has_building = building_university
						any_scope_building = {
							is_building_type = building_university
							is_under_construction = no
						}
					}
				}
			}
		}
	}
}
